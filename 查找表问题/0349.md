class Solution:
    def intersection(self, nums1, nums2):
        # 让 num1 是元素多的那个数组
        if len(nums1) < len(nums2):
            nums1, nums2 = nums2, nums1
        s = set(nums1)
        return list({x for x in nums2 if x in s})
